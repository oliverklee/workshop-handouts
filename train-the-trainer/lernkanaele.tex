\section{Effektivität von Lernkanälen}
\subsection{Wahrnehmen (Bandbreite)}
Von wenig zu viel:
\begin{enumerate}
  \item riechen, schmecken oder fühlen
  \item hören
  \item sehen
\end{enumerate}

\subsection{Behalten}
Von wenig zu viel:
\begin{enumerate}
  \item lesen
  \item hören
  \item sehen
  \item hören und sehen
  \item sagen
  \item selber machen
  \item es anderen erklären/beibringen
\end{enumerate}
